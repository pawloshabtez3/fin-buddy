### âš™ï¸ Gemini API Example (lib/gemini.ts)
```ts
import { GoogleGenerativeAI } from "@google/generative-ai";

const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);

export async function getSpendingInsights(expenses: any) {
  const prompt = `Analyze this user's spending data: ${JSON.stringify(expenses.slice(-10))}\n\nSummarize patterns, suggest two savings tips, and end with a motivational message.`;
  
  const model = genAI.getGenerativeModel({ model: "gemini-1.5-pro" });
  const result = await model.generateContent(prompt);
  return result.response.text();
}
```
